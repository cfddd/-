const int N = 110;

int a[N];
bool solve() 
{
    int n;cin >> n;
    for(int i = 0;i < n;i ++)cin >> a[i];
    sort(a,a+n);

    for(int i = 1;i < n;i ++)
        if(a[i] == a[i-1])
        {
            return 0;
        }
    
    
    bool acc = 1;
    for(int i = 2;i <= n;i ++)
    {
        vector<int> num(i,0);
        for(int j = 0;j < n;j ++)num[a[j] % i]++;
        if(*min_element(num.begin(),num.end()) >= 2)acc = 0;
    }
    return acc;
    
}
